import cv2 import base64 import time import threading import numpy as np import onnxruntime as ort from flask import Flask, render_template, Response, request, jsonify from deepface import DeepFace app = Flask(__name__) # ===================== 固定配置 ===================== IP_CAMERA_URL = "http://10.215.158.45:8080/video" # 仅保留IP摄像头 ONNX_MODEL_PATH = r"C:\Users\purriste\Desktop\PYProject\rppg\backend\core\models\emotion-ferplus-8.onnx" # 情绪标签（FER+ 8类） EMOTION_LABELS = ["neutral", "happiness", "surprise", "sadness", "anger", "disgust", "fear", "contempt"] # 全局识别方法切换（deepface / onnx） current_method = "deepface" # 加载ONNX模型 ort_session = ort.InferenceSession(ONNX_MODEL_PATH) # ===================== 核心情绪处理器（完全按你的参考代码重写） ===================== class EmotionProcessor: def __init__(self, video_url): self.cap = cv2.VideoCapture(video_url) self.cap.set(cv2.CAP_PROP_BUFFERSIZE, 1) self.raw_frame = None self.latest_result = {"emotion": "neutral", "confidence": 0, "region": None} self.running = True # 双线程：采集 + 分析 threading.Thread(target=self._capture_loop, daemon=True).start() threading.Thread(target=self._analysis_loop, daemon=True).start() def _capture_loop(self): """视频帧采集线程""" while self.running: ret, frame = self.cap.read() if ret: self.raw_frame = frame else: self.cap.release() self.cap = cv2.VideoCapture(IP_CAMERA_URL) time.sleep(0.01) def _analysis_loop(self): """情绪分析线程（支持DeepFace + ONNX双方法）""" global current_method while self.running: if self.raw_frame is None: time.sleep(0.1) continue try: # ============= DeepFace ============= if current_method == "deepface": res = DeepFace.analyze( self.raw_frame, actions=['emotion'], enforce_detection=False, detector_backend='mediapipe', silent=True ) if res: r = res[0] emo = r['dominant_emotion'] conf = int(r['emotion'][emo]) self.latest_result = { "emotion": emo, "confidence": conf, "region": r['region'] } # ============= ONNX-FER+ ============= elif current_method == "onnx": gray = cv2.cvtColor(self.raw_frame, cv2.COLOR_BGR2GRAY) face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + "haarcascade_frontalface_default.xml") faces = face_cascade.detectMultiScale(gray, 1.1, 4) if len(faces) > 0: x, y, w, h = faces[0] face_gray = gray[y:y + h, x:x + w] # ONNX 预处理 face = cv2.resize(face_gray, (64, 64)) face = face.astype(np.float32) / 255.0 face = np.expand_dims(face, axis=(0, 1)) # [1,1,64,64] out = ort_session.run(None, {"input": face}) idx = np.argmax(out[0][0]) emo = EMOTION_LABELS[idx] conf = int(float(out[0][0][idx]) * 100) self.latest_result = { "emotion": emo, "confidence": conf, "region": {"x": x, "y": y, "w": w, "h": h} } except Exception: pass time.sleep(0.6) def get_ui_frame(self): """获取用于前端展示的帧（base64 + 情绪结果）""" if self.raw_frame is None: return None frame = self.raw_frame.copy() res = self.latest_result # 画框（橙色主题色） if res.get("region"): r = res["region"] x, y, w, h = r['x'], r['y'], r['w'], r['h'] color = (34, 114, 255) # FF7222 橙色 cv2.rectangle(frame, (x, y), (x + w, y + h), color, 3) # 标注情绪 text = f"{res['emotion']} ({res['confidence']}%)" cv2.putText(frame, text, (x, y - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.9, color, 2) # 压缩为JPEG _, buffer = cv2.imencode('.jpg', frame, [cv2.IMWRITE_JPEG_QUALITY, 80]) return { "image": base64.b64encode(buffer).decode('utf-8'), "emotion": res["emotion"], "confidence": res["confidence"] } def stop(self): self.running = False self.cap.release() # ===================== 全局初始化 ===================== processor = EmotionProcessor(IP_CAMERA_URL) # ===================== Flask 接口 ===================== @app.route('/') def index(): return render_template('emo.html') @app.route('/switch_method', methods=['POST']) def switch_method(): global current_method method = request.json.get('method') if method in ["deepface", "onnx"]: current_method = method return jsonify(status="ok", method=current_method) def gen_video(): while True: data = processor.get_ui_frame() if data is None: time.sleep(0.1) continue yield f"data: {data['image']}\n\n" time.sleep(0.03) @app.route('/video_feed') def video_feed(): return Response(gen_video(), mimetype='text/event-stream') if __name__ == '__main__': app.run(host='0.0.0.0', port=5000, debug=True, threaded=True)
